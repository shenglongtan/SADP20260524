def find_latest_run_folder(root: Path) -> Path:
    candidates = []
    for p in root.rglob('Parameters.json'):
        candidates.append(p.parent)
    if not candidates:
        raise FileNotFoundError(f'No Parameters.json found under: {root}')
    candidates = sorted(candidates, key=lambda x: x.stat().st_mtime, reverse=True)
    return candidates[0]

def compute_anomaly_score(y_true: np.ndarray, y_pred: np.ndarray, horizon_reduce: str = 'max', var_reduce: str = 'max'):
    """
    计算窗口级异常分值。

    y_true / y_pred: [N, V, H]
    horizon_reduce: 对 horizon 维聚合方式，支持 'max' 或 'mean'
    var_reduce: 对变量维聚合方式，支持 'max' 或 'mean'
    """
    if y_true.shape != y_pred.shape:
        raise ValueError('y_true and y_pred must have the same shape.')

    error = np.abs(y_true - y_pred)

    if horizon_reduce == 'max':
        var_error = error.max(axis=2)
    elif horizon_reduce == 'mean':
        var_error = error.mean(axis=2)
    else:
        raise ValueError("horizon_reduce must be 'max' or 'mean'")

    mu = var_error.mean(axis=0, keepdims=True)
    sigma = var_error.std(axis=0, keepdims=True)
    sigma = np.where(sigma == 0, 1e-8, sigma)
    norm_var_error = (var_error - mu) / sigma

    if var_reduce == 'max':
        score = norm_var_error.max(axis=1)
    elif var_reduce == 'mean':
        score = norm_var_error.mean(axis=1)
    else:
        raise ValueError("var_reduce must be 'max' or 'mean'")

    return {
        'error_tensor': error,
        'var_error': var_error,
        'norm_var_error': norm_var_error,
        'score': score,
        'mu': mu.squeeze(0),
        'sigma': sigma.squeeze(0),
    }

result = compute_anomaly_score(y_true, y_pred, horizon_reduce='max', var_reduce='max')
score = result['score']

def choose_threshold(scores: np.ndarray, method: str = 'percentile', percentile: float = 99.0):
    if method == 'percentile':
        return float(np.percentile(scores, percentile))
    elif method == 'mean_std':
        return float(scores.mean() + 3 * scores.std())
    else:
        raise ValueError("method must be 'percentile' or 'mean_std'")

threshold = choose_threshold(score, method='percentile', percentile=99.0)
pred_label = (score > threshold).astype(int)

print('threshold =', threshold)
print('anomaly count =', int(pred_label.sum()))
print('anomaly ratio =', float(pred_label.mean()))
